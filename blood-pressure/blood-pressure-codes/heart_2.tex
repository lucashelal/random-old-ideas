\begin{document}
	%
	\tdplotsetmaincoords{45}{45}
	\begin{tikzpicture}[tdplot_main_coords,scale=2.0]
		\pgfmathsetmacro{\tini}{0.5*pi}
		\pgfmathsetmacro{\tfin}{-0.5*pi}
		\pgfmathsetmacro{\tend}{2.5*pi}
		%%% Coordinate axis
		\draw[thick,->] (2.3,-0.20,0.8) -- (5.5,0.05,0.8) node [below left] {\footnotesize$x$};
		\draw[dashed] (1,-0.20,0.5) -- (2.5,0,0.5);
		\draw[thick,->] (3.4,-2,1) -- (3.4,1.8,1) node [right] {\footnotesize$y$};
		\draw[dashed] (3.4,0,0) -- (3.4,0,-1);
		\draw[thick,->] (3.4,0,0) -- (3.4,0,3.5) node[above] {\footnotesize$z$};

       
  
		% The paraboloid (for z = constant)
		\foreach \altura in {0.0125,0.025,...,1}{
			\pgfmathparse{sqrt(\altura)}
			\pgfmathsetmacro{\radio}{\pgfmathresult}
			\draw[black,thick,opacity=1] plot[domain=\tini:\tfin,smooth,variable=\t] ({2+\radio*1.8*cos(\t r)},{\altura-0.2},{\radio*sin(\t r-1)}); 
        \draw[gray!90!black!70,thick,fill=white!70!black!10,opacity=1] plot[domain=\tini:\tfin,smooth,variable=\t] ({2+\radio*1.8*cos(\t r)},{\altura-0.2},{\radio*sin(\t r-1)}); 
		}

     \draw[ultra thick,->] (1,-0.3,0.8) -- (4.5,1.5,-1);

	\end{tikzpicture}
